You can mod your own cassette with your own song.

- Use OGG files for sound!
- Replace <Your_Unique_Cassette_Name_Here> by your own value, if you dont, the cassette will dont work.
- Dont forget it's not recommended to publish copryrighted songs without autorization.

class CfgPatches
{
	class <What_You_Want_Here>
	{
		units[] = {};
		weapons[] = {};
		requiredVersion = 0.1;
		requiredAddons[] = 
		{
			"DZ_Data", 
			"DZ_Sounds_Effects",
			"HRS_Items",
			"HRS_Items_cfg",
			"HRS_Scripts"
		};
	};
};

class CfgVehicles
{
	class HRS_K7;
	class HRS_K7_<Your_Unique_Cassette_Name_Here>: HRS_K7
	{
		scope = 2;
		displayName = "";
		descriptionShort = "";
	};
};


class CfgSoundShaders
{
	class HRS_K7_SoundShader;
	class HRS_K7_<Your_Unique_Cassette_Name_Here>_SoundShader: HRS_K7_SoundShader
	{
		samples[]=
		{	
			{"path/to/your/ogg/sound/file",1}
		};
		volume=1.0;
	};		
};

class CfgSoundSets
{
	class HRS_K7_<Your_Unique_Cassette_Name_Here>_SoundSet
	{
		soundShaders[]=
		{
			"HRS_K7_<Your_Unique_Cassette_Name_Here>_SoundShader"
		};
	};
};
